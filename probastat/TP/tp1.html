def dif(A):
    B=[A[i] for i in range(len(A))]
   
    for n in range(1,len(A)):
        
        A[n]=A[n]-B[n-1]
    return A

def pascal(n):
    matrix=np.zeros((n+1,n+1))
   
    for i in range(n+1):
        if i==0:
            matrix[i,0]=1
        else:

            for j in range(i+1):
                if j==0:
                    matrix[i,0] =1
                else:
                    matrix[i,j]=matrix[i-1, j-1] + matrix[i-1, j]
    return matrix

def vandermonde(X):
    n=len(X)
    A=np.ones((n,n))
    for i in range(1, n):
        A[i]=np.array(X)**i
    return A, np.linalg.det(A)
