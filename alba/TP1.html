def recgcd(a, b):
    if b==0:
        return a
    else:
        return recgcd(b, a%b)

def itgcd(a,b):
    while b>1:
        r=a%b
        a=b
        b=r
    return a

def SelectionSort(L):
    for i in range(len(L)-1):
        jMin = i
        for j in range(i+1, len(L)):
            if L[j] < L[jMin]:
                jMin=j
        if (jMin !=i):
            L[jMin], L[i] = L[i], L[jMin]
    return L
